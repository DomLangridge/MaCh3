188inline double GetBetaParameter(const double data, const double mc, const double w2, TestStatistic TestStat) {
189// ****************
190 double Beta = 0.0;
191
192 if (TestStat == kDembinskiAbdelmottele) {
193 //the so-called effective count
194 const double k = mc*mc / w2;
195 //Calculate beta which is scaling factor between true and generated MC
196 Beta = (data + k) / (mc + k);
197 }
198 //KS: Below is technically only true for Cowan's BB, which will not be true for Poisson or IceCube, because why not...
199 else {
200 // CW: Barlow-Beeston uses fractional uncertainty on MC, so sqrt(sum[w^2])/mc
201 const double fractional = std::sqrt(w2)/mc;
202 // CW: -b/2a in quadratic equation
203 const double temp = mc*fractional*fractional-1;
204 // CW: b^2 - 4ac in quadratic equation
205 const double temp2 = temp*temp + 4*data*fractional*fractional;
206 if (temp2 < 0) {
207 std::cerr << "Negative square root in Barlow Beeston coefficient calculation!" << std::endl;
208 std::cerr << __FILE__ << ":" << __LINE__ << std::endl;
209 throw;
210 }
211 // CW: Solve for the positive beta
212 Beta = (-1*temp+std::sqrt(temp2))/2.;
213 }
214 return Beta;
215}

220inline double GetSubOptimality(std::vector<double> EigenValues, const int TotalTarameters) {
221// *********************
222 double sum_eigenvalues_squared_inv = 0.0;
223 double sum_eigenvalues_inv = 0.0;
224 for (unsigned int j = 0; j < EigenValues.size(); j++)
225 {
226 //KS: IF Eigen values are super small skip them
227 //if(EigenValues[j] < 0.0000001) continue;
228 sum_eigenvalues_squared_inv += std::pow(EigenValues[j], -2);
229 sum_eigenvalues_inv += 1.0 / EigenValues[j];
230 }
231 const double SubOptimality = TotalTarameters * sum_eigenvalues_squared_inv / std::pow(sum_eigenvalues_inv, 2);
232 return SubOptimality;
233}
