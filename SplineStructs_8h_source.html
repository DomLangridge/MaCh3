114 if(
115 (XsecCov->GetParSplineKnotUpperBound(splineParsIndex) != 9999)

117 {
118 for(int knotId = 0; knotId < xsecgraph->GetN(); knotId++){
119 double x,y;
120

122 if( xsecgraph->GetPoint(knotId, x, y) == -1) {

124 throw MaCh3Exception(__FILE__ , __LINE__ );

128 y = std::max(y, XsecCov->GetParSplineKnotLowerBound(splineParsIndex));
129
130 xsecgraph->SetPoint(knotId, x, y);
131 }
132
133 // EM: check if our cap made the spline flat, if so we set to 1 knot to avoid problems later on
134 bool isFlat = true;
135
136 for(int knotId = 0; knotId < xsecgraph->GetN(); knotId++){
137 double x,y;
138

579 inline int FindX(double x) {
580 // The segment we're interested in (klow in ROOT code)
581 int segment = 0;
582 int kHigh = nPoints-1;
583 // If the variation is below the lowest saved spline point
584 if (x <= XPos[0]){
585 segment = 0;
586 // If the variation is above the highest saved spline point
587 } else if (x >= XPos[nPoints-1]) {
588 // Yes, the -2 is indeed correct, see TSpline.cxx:814 and //see: https://savannah.cern.ch/bugs/?71651
589 segment = kHigh;
590 // If the variation is between the maximum and minimum, perform a binary search
591 } else {
592 // The top point we've got
593 int kHalf = 0;
594 // While there is still a difference in the points (we haven't yet found the segment)
595 // This is a binary search, incrementing segment and decrementing kHalf until we've found the segment
596 while (kHigh - segment > 1) {
597 // Increment the half-step
598 kHalf = (segment + kHigh)/2;
599 // If our variation is above the kHalf, set the segment to kHalf
600 if (x > XPos[kHalf]) {
601 segment = kHalf;
602 // Else move kHigh down
603 } else {
604 kHigh = kHalf;
605 }
606 } // End the while: we've now done our binary search
607 } // End the else: we've now found our point
608 if (segment >= nPoints-1 && nPoints > 1) segment = nPoints-2;
609 return segment;
610 }

613 inline double Eval(double var) {
614 // Get the segment for this variation
615 int segment = FindX(var);
616 // The get the coefficients for this variation
617 _float_ x = -999.99, y = -999.99, b = -999.99, c = -999.99, d = -999.99;
618 GetCoeff(segment, x, y, b, c, d);
619 double dx = var - x;
620 // Evaluate the third order polynomial
621 double weight = y+dx*(b+dx*(c+d*dx));
622 return weight;
623 }

684 inline int FindX(double x) {
685 // The segment we're interested in (klow in ROOT code)
686 int segment = 0;
687 int kHigh = nPoints-1;
688 // If the variation is below the lowest saved spline point
689 if (x <= XPos[0]){
690 segment = -1;
691 // If the variation is above the highest saved spline point
692 } else if (x >= XPos[nPoints-1]) {
693 segment = -2;
694 // If the variation is between the maximum and minimum, perform a binary search
695 } else {
696 // The top point we've got
697 int kHalf = 0;
698 // While there is still a difference in the points (we haven't yet found the segment)
699 // This is a binary search, incrementing segment and decrementing kHalf until we've found the segment
700 while (kHigh - segment > 1) {
701 // Increment the half-step
702 kHalf = (segment + kHigh)/2;
703 // If our variation is above the kHalf, set the segment to kHalf
704 if (x > XPos[kHalf]) {
705 segment = kHalf;
706 // Else move kHigh down
707 } else {
708 kHigh = kHalf;
709 }
710 } // End the while: we've now done our binary search
711 } // End the else: we've now found our point
712 if (segment >= nPoints-1 && nPoints > 1) segment = nPoints-2;
713 return segment;
714 }

717 inline double Eval(double var) {
718 // Get the segment for this variation
719 int segment = FindX(var);
720 // The get the coefficients for this variation
721 _float_ x = -999.99, y = -999.99, b = -999.99, c = -999.99, d = -999.99;
722
723 if(segment >=0){
724 GetCoeff(segment, x, y, b, c, d);
725 }
726
727 // if var is outside of the defined range, set the coefficients to 0 so that Eval just returns the value at the end point of the spline
728 else if(segment == -1){
729 GetKnot(0, x, y);
730 b = 0.0;
731 c = 0.0;
732 d = 0.0;
733 }
734 else if(segment == -2){
735 GetKnot(nPoints-1, x, y);
736 b = 0.0;
737 c = 0.0;
738 d = 0.0;
739 }
740
741 double dx = var - x;
742 // Evaluate the third order polynomial
743 double weight = y+dx*(b+dx*(c+d*dx));
744 return weight;
745 }
